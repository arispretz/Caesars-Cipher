function rot13(str) {
let decoded = '';
for (let i = 0; i < str.length; i++) {
let letterCode = str[i].charCodeAt();
  if (letterCode >= 65 && letterCode <= 77) {
    decoded += String.fromCharCode(letterCode + 13);
} else if (letterCode >= 78 && letterCode <= 90) {
    decoded += String.fromCharCode(letterCode - 13);
} else {
    decoded += str[i];
}
}
  return decoded;
}


rot13("SERR PBQR PNZC");
